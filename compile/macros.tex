\global\Formula=0
\global\Section=0
\global\advance\Chapter by 1
%\johanlabel{#1}%

\global\Formula=0
\global\advance\Section by 1
%\johanlabel{#1}%
\the\Chapter.\the\Section
\immediate\write\CROSSREFERENCES{\letterbackslash def\letterbackslash%
oops#1{\the\Chapter.\the\Section}}\ignorespaces}
